\section{True SGD}

\subsection{Setup}

$f(x) = \frac{1}{2} \|\Phi x - y\|^2$, $H = \Phi^\top \Phi$

Let $x_*$ be a solution, $Hx_* = \Phi^\top y$

\bigskip

$\forall t, \quad y_t = \phi_t^\top x_* + \epsilon_t$ with 
\[\left\{
\begin{array}{l}
\epsilon _t \sim \mathcal N(0, \sigma^2) \text{ i.i.d.}\\
\phi_t \sim \mathcal N(0, H) \text{ i.i.d.}\\
\epsilon_t \text{ and } \phi_t \text{ independent}
%\E(\phi_t y_t) = \Phi^\top y
\end{array}\right.
\]

Hence $\E[\phi_t y_t] = \E[\phi_t(\phi_t^\top x_* + \epsilon_t)] = Hx_* + \E[\phi_t \epsilon_t] = \Phi^\top y$

\bigskip

Consider a stochastic gradient
\[
g_t = \phi_t(\phi_t^\top x_t - y_t) = \phi_t \phi_t^\top (x_t - x_*) -\epsilon_t  \phi_t 
\]

Then the stochastic gradient descent (SGD) is given by
\[
\forall t \geq 0, \quad x_{t+1}-x_* = x_t - \eta_t g_t - x_* =  (I-\eta_t \phi_t \phi_t^\top)(x_t-x_*) + \eta_t \epsilon_t \phi_t
\]


\subsection{Calculation of the Risk}

In this section, we find a closed form of the excess risk $\E[f(x_t)-f_*]$ as a function of $(\eta_t)$.

\bigskip

Let $Q$ s.t. $H = Q\Lambda Q^\top$ with $\Lambda = \diag(\lambda)$,\\
$M_t= Q^\top \E[u_tu_t^\top] Q $

\bigskip


Therefore $\E[f(x_t)-f_*] = \frac{1}{2}\E[\langle u_t, Hu_t\rangle] = \frac{1}{2}\tr(Hu_tu_t^\top)=\frac{1}{2}\langle M_t, \Lambda \rangle =\frac{1}{2}\langle  m_t , \lambda \rangle$

With $m_t$ the diagonal of $M_t$.

We will use the following:

\begin{lemma} (A corollary of Isserlis's theorem). If $X\sim \mathcal N(0,H)$ and $M \in \R^{n\times n}$ is a constant matrix,
\[
\E[XX^\top M XX^\top] = \tr(MH)H + HMH+ H M^\top H
\]
\end{lemma}

\begin{align*}
M_{t+1} &= Q^\top \E[((I-\eta_t \phi_t \phi_t^\top)u_t + \eta_t \epsilon_t \phi_t)((I-\eta_t \phi_t \phi_t^\top)u_t + \eta_t \epsilon_t \phi_t)^\top] Q\\
&= Q^\top (\E[(I-\eta_t \phi_t \phi_t^\top)u_tu_t^\top (I-\eta_t \phi_t \phi_t^\top)]+\eta_t^2\sigma^2H)Q\\
&= Q^\top \E[ (u_t-\eta_t\phi_t\phi_t^\top u_t)(u_t-\eta_t\phi_t\phi_t^\top u_t)^\top] Q + \eta_t^2\sigma^2\Lambda \qquad (\text{$\epsilon_t$ and $\phi_t$ independent and $\E\epsilon_t =0$})\\
&= M_t - \eta_t \Lambda M_t - \eta_t M_t \Lambda + \eta_t^2 Q^\top \E[\phi_t \phi_t^\top u_t u_t^\top \phi_t\phi_t^\top]Q + \eta_t^2\sigma^2\Lambda\\
&=  M_t - \eta_t \Lambda M_t - \eta_t M_t \Lambda + \eta_t^2 (2 \Lambda M_t \Lambda + \tr(\Lambda M_t)\Lambda) + \eta_t^2\sigma^2\Lambda \quad \text{($u_t$ and $\phi_t$ independent and $\phi_t$ is gaussian)}
\end{align*}


Hence, $m_{t+1} =  m_t - 2\eta_t \Lambda m_t + 2\eta_t^2  \Lambda^2 m_t + \eta_t^2 \lambda \lambda^\top m_t + \eta_t^2\sigma^2\lambda = \left[(I - \eta_t\Lambda)^2 + \eta_t^2 \Lambda^2 + \eta_t^2\lambda\lambda^\top \right] m_t + \eta_t^2 \sigma^2 \lambda$

Let $A_t = (I - \eta_t\Lambda)^2 + \eta_t^2 \Lambda^2+\eta_t^2\lambda\lambda^\top $ and $A_{s:t} = A_t...A_s$

\bigskip 
Then $\displaystyle m_t = A_{0:{t-1}}m_0 + \sum_{k=0}^{t-1} A_{k+1:t-1}\eta_k^2\sigma^2\lambda$
\[
\E[f(x_t) - f_*] = \frac{1}{2}\lambda^\top \left( \prod_{k=0}^{t-1}[(I - \eta_k\Lambda)^2 + \eta_k^2 \Lambda^2+\eta_k^2\lambda\lambda^\top]m_0 + \sum_{k=0}^{t-1}\eta_k^2\sigma^2 \prod_{s=k+1}^{t-1}[(I - \eta_s\Lambda)^2 + \eta_s^2 \Lambda^2+\eta_s^2\lambda\lambda^\top]\lambda  \right)
\]


\subsection{Approximation}

Replace $\lambda \lambda^\top$by $\Lambda^2$. $\lambda \lambda^\top$ and $\Lambda$ have the same diagonal, and this change will make the matrices $A_t$ commutative.

Suppose $\lambda_i = Li^{-\alpha}$
\[
\E[f(x_t) - f_*] \approx \frac{1}{2} \sum_{i=1}^d Li^{-\alpha} \left( m_{0,i} \prod_{j=0}^{t-1} \left[ (1-\eta_j Li^{-\alpha})^2+2\eta_j^2 L^2 i^{-2\alpha} \right] + \sum_{k=0}^{t-1} \eta_k^2 \sigma^2 Li^{-\alpha} \prod_{j=k+1}^{t-1} \left[ (1-\eta_j Li^{-\alpha})^2+2\eta_j^2 L^2 i^{-2\alpha} \right] \right)
\]

\subsubsection{If  $\eta_k = \eta$ is constant}

We want to model the risk as a simple function of $t$.

\begin{equation}\label{eq:approx}
\E[f(x_t) - f_*] \approx \frac{1}{2} \sum_{i=1}^d Li^{-\alpha}\left(\left[(1-\eta Li^{-\alpha})^2+2\eta^2L^2i^{-2\alpha} \right]^t m_{0,i} + \sum_{k=0}^{t-1} \eta^2 \sigma^2 Li^{-\alpha} \left[(1-\eta Li^{-\alpha})^2+2\eta^2L^2i^{-2\alpha}\right]^{t-1-k}\right)
\end{equation}

$\displaystyle\sum_{k=0}^{t-1} \eta^2 \sigma^2 Li^{-\alpha} \left[(1-\eta Li^{-\alpha})^2+2\eta^2L^2i^{-2\alpha}\right]^{t-1-k}$ is a geometric sum, therefore 
\[
\sum_{k=0}^{t-1} \eta^2 \sigma^2 Li^{-\alpha} \left[(1-\eta Li^{-\alpha})^2+2\eta^2L^2i^{-2\alpha}\right]^{t-1-k} = \eta^2\sigma^2Li^{-\alpha} \frac{1 - \left[(1-\eta Li^{-\alpha})^2+2\eta^2L^2i^{-2\alpha}\right]^t}{1 - \left[(1-\eta Li^{-\alpha})^2+2\eta^2L^2i^{-2\alpha}\right]}
\]

We can approximate this sum with an integral, supposing $d \to \infty$ and $m_{0,i} \sim \Delta/i^\beta$
\[
\mathbb{E}[f(x_t) - f_*] \approx \frac{1}{2} \int_{1}^\infty L u^{-\alpha} \left( \left[ (1-\eta L u^{-\alpha})^2+2\eta^2 L^2 u^{-2\alpha} \right]^t \Delta u^{-\beta} + \eta^2\sigma^2Lu^{-\alpha} \frac{1 - \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]^t}{1 - \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]}\right) du
\]


$\bullet$ Bias term: $b = \frac{1}{2} \int_1^\infty L u^{-\alpha}  \left[ (1-\eta L u^{-\alpha})^2+2\eta^2 L^2 u^{-2\alpha} \right]^t \Delta u^{-\beta} du$

\begin{align*}
b= \frac{L\Delta}{2} \int_1^\infty \frac{1}{u^{\alpha+\beta} }e^{t \ln((1-\eta L u^{-\alpha})^2+2\eta^2 L^2 u^{-2\alpha})}  du
\end{align*}

Moreover, 

$\ln((1-\eta L u^{-\alpha})^2+2\eta^2 L^2 u^{-2\alpha}) =  \ln(1 -2\eta L u^{-\alpha}+3\eta^2 L^2 u^{-2\alpha}) = -2\eta L u^{-\alpha} +(\eta Lu^{-\alpha})^2 + O(u^{-3\alpha})$

\begin{align*}
b &\approx \frac{L\Delta}{2} \int_1^\infty \frac{1}{u^{\alpha+\beta} }e^{  (-2\eta L u^{-\alpha} )t}  du, \qquad \text{Let }z=u^{-\alpha},\; du =-\frac{1}{\alpha}z^{-\frac{\alpha+1}{\alpha}}dz\\
&\approx \frac{L\Delta}{2} \int_0^1 \frac{z^{1+\frac{\beta}{\alpha}-\frac{1+\alpha}{\alpha}}}{\alpha }e^{  -2\eta Lz t}  dz\\
&\approx \frac{L\Delta}{2\alpha} \int_0^1 z^{\frac{\beta-1}{\alpha}}e^{  -2\eta Lz t}  dz\\
&\sim_{t\to \infty} \frac{L\Delta}{2\alpha} \frac{\Gamma(\frac{\beta-1}{\alpha}+1)}{(2\eta L t)^{\frac{\beta-1}{\alpha}+1}} \qquad \text{(Watson's lemma)}
\end{align*}


$\bullet$ Variance term: $V = \int_{1}^\infty L u^{-\alpha}\times \eta^2\sigma^2Lu^{-\alpha} \frac{1 - \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]^t}{1 - \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]}du$

\begin{align*}
V &= \frac{1}{2} \int_{1}^\infty \frac{L^2 \eta^2\sigma^2}{u^{2\alpha}} \frac{1 - \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]^t}{1 - \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]}du
\end{align*}
We make the rough approximations:
$\left\{
\begin{array}{l}
 \left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right] \approx 1 - 2\eta L u^{-\alpha}\\
 1-\left[(1-\eta Lu^{-\alpha})^2+2\eta^2L^2u^{-2\alpha}\right]^{t} \approx 1
\end{array}
\right.$

Then
\begin{align*}
V&\approx \frac{1}{2}  \int_{1}^\infty \frac{L^2 \eta^2\sigma^2}{u^{2\alpha}} \frac{1 }{ 2\eta Lu^{-\alpha}}du\\
&\approx \frac{1}{2} \int_{1}^\infty \frac{L \eta\sigma^2}{2u^{\alpha}} du\\
&\approx \frac{L \eta\sigma^2}{4(\alpha-1)} \qquad \text{if } \alpha>1
\end{align*}

Total approximation (Diagonal and Integral) if $\alpha>1$:
\[
b + V =  \frac{L\Delta}{2\alpha} \frac{\Gamma(\frac{\beta-1}{\alpha}+1)}{(2\eta L t)^{\frac{\beta-1}{\alpha}+1}} + \frac{L \eta\sigma^2}{4(\alpha-1)} 
\]

\subsubsection{WSD schedule}

\begin{align*}
b_t &= \frac{1}{2} \int_{1}^{\infty} \frac{L\Delta}{u^{\alpha+\beta}} \prod_{j=0}^{t-1} \left[ \left( 1 - \eta_j \frac{L}{u^\alpha} \right)^2 + 2\eta_j \frac{L^2}{u^{2\alpha}} \right] du\\

\[\begin{array}{lllll}

\end{array}\]

The loss function is quadratic \[L(\hat y,y) = \frac{1}{2} \|\hat y-y \|^2\]


For a given $(x,y)$, we want to minimize \[f_{x,y}:W \mapsto L(W_2W_1x, y) = \frac{1}{2} \|W_2W_1 x - y\|^2 \]

The theoretical risk is $R(W) = \E_{x,y}[f_{x,y}(W)]$

\bigskip

We suppose $x \sim \mathcal N(0, H)$ and $y = W_2^*W_1^*x+\epsilon$ with $\epsilon \sim \mathcal N(0,\Sigma)$ independent of $x$.



Therefore the risk is 
\begin{align*}
R(W)  &= \E[f_{x,y}(W)]
\\&=
\E\left[\frac{1}{2} \|W_2W_1 x - W_2^*W_1^*x-\epsilon\|^2 \right]
\\&=
\E\left[\frac{1}{2} \tr \big[((W_2W_1 - W_2^*W_1^*)x-\epsilon)((W_2W_1 - W_2^*W_1^*)x-\epsilon)^\top \big] \right]

\\&= \frac{1}{2}\tr \big[(W_2W_1 - W_2^*W_1^*)H (W_2W_1 - W_2^*W_1^*)^\top ] + \frac{1}{2} \tr\Sigma
\end{align*}
