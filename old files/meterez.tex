\subsection{Setup}


\begin{itemize}

\item[$-$] $x \sim \mathcal N(0,H)$

\item[$-$] $y=x^\top w^* + \epsilon$ \quad with $\E[\epsilon]=0, \var(\epsilon) = \sigma^2$

\item[$-$] $w_{t+1} = w_t - \eta_t x_t (x_t^\top w_t - y_t)$, \quad $y_t = x_t^\top w^\star +\epsilon_t$

\item[$-$] $\mathcal R(w) =  \frac{1}{2}\E_{x,y}[(x^\top w - y)^2]$


\item[$-$] $H = Q\Lambda Q^\top$ \quad and $\Lambda = \mathrm{Diag(\lambda)}$

\item[$-$] $\Sigma_t = \E[(w_t - w^\star)(w_t - w^\star)^\top]$ and $M_t = Q \Sigma_t Q^\top$

\end{itemize}

\begin{align*}
\E [x^\top (w - w^*) (w-w^*)^\top x]
&= \E [x^\top (w - w^*) (w-w^*)^\top x]\\
&= \E[\tr( (w - w^*) (w-w^*)^\top x x^\top)]\\
&= \tr\left(\E[(w - w^*) (w-w^*)^\top] \E[xx^\top]\right)\\
&=\langle \Sigma,H  \rangle\\
&= \langle M,\Lambda  \rangle 
\end{align*}

Then \[\mathcal R(w_t) - \frac{1}{2}\sigma^2 = \frac{1}{2}\langle m_t, \lambda\rangle \qquad \text{with } m_t = \mathrm{diag}(M_t) \]

\subsection{New Variance}

\begin{align*}
\sum_{p=0}^{T/2} \frac{1}{p^{2\gamma}} \exp\left(-\eta \lambda_k(T^{1-\gamma} - p^{1-\gamma})\right) 
&\leq  \sum_{p=0}^{T/2} \frac{1}{p^{2\gamma}} \exp\left(-(1-\gamma)\eta \lambda_kT^{-\gamma}(T- p)\right) \\
& \leq \sum_{p=0}^{T/2} \frac{1}{p^{2\gamma}} \exp\left(-\frac{1}{2}(1-\gamma)\eta \lambda_kT^{1-\gamma}\right) \\
& \leq \exp\left(-\frac{1}{2}a_kT^{1-\gamma} \right) s_T(\gamma)
\end{align*}

With $a_k := (1-\gamma) \eta \lambda_k$ and $s_T(\gamma) = \left\{ \begin{array}{ll}
T^{1-2\gamma}  & \gamma < 1/2\\
\ln(T) & \gamma = 1/2\\
1 & \gamma >1/2
\end{array}\right.$


\begin{align*}
\sum_{p=T/2}^{T} \frac{1}{p^{2\gamma}} \exp\left(-\eta \lambda_k(T^{1-\gamma} - p^{1-\gamma})\right) 
&\leq  \sum_{p=T/2}^{T} \frac{1}{p^{2\gamma}} \exp\left(-a_kT^{-\gamma}(T- p)\right) \\
& \leq \left(\frac{T}{2}\right)^{-2\gamma} \exp\left(-a_kT^{1-\gamma}\right)\sum_{p=T/2}^{T}  \exp\left(a_kT^{-\gamma}p\right) \\
& \leq \left(\frac{T}{2}\right)^{-2\gamma} \exp\left(-a_kT^{1-\gamma}\right) \left[ \exp\left(a_kT^{1-\gamma}\right) - \exp\left(\frac{1}{2}a_kT^{1-\gamma}\right) \right]\frac{1}{a_kT^{-\gamma}} \\
& \leq \left(\frac{T}{2}\right)^{-2\gamma}  \left[ 1-\exp\left(-\frac{1}{2}a_kT^{1-\gamma}\right)  \right] \frac{1}{a_kT^{-\gamma}}\\
& \leq T^{-\gamma} 2^{2\gamma} a_k^{-1} \left[ 1-\exp\left(-\frac{1}{2}a_kT^{1-\gamma}\right)  \right] 
\end{align*}

\[
variance_{T+1} = \sum_k \eta \sigma^2 \lambda_k^2 \sum_{p=0}^T \frac{1}{p^{2\gamma}} \exp\left(-2\eta \lambda_k(T^{1-\gamma} - p^{1-\gamma})\right)  
\]

Therefore

\begin{align*}
 variance_{T+1} &\leq \sum_k \eta \sigma^2 \lambda_k^2 \left(  \exp\left(-\frac{1}{2}a_kT^{1-\gamma} \right) s_T(\gamma)+\frac{2^{2\gamma}T^{-\gamma}}{ a_k}\left[ 1-\exp\left(-\frac{1}{2}a_kT^{1-\gamma}\right)  \right]\right)\\
& \lesssim \sum_k \eta \sigma^2\lambda_k^2 T^{1-2\gamma}\exp\left(-\frac{1}{2}(1-\gamma)\eta\lambda_k T^{1-\gamma}\right) + \sum_k  \frac{\sigma^2\lambda_k}{1-\gamma }  T^{-\gamma}
\end{align*}

\[
bias_{T+1} = \sum_k m_{0k}\lambda_k \exp(-\eta \lambda_k T^{1-\gamma}) 
\]


Let $k^* := \max\{k;\lambda_k \geq \frac{\ln T}{\eta T^{1-\gamma}}\}$

\begin{align*}
bias_{T+1} &= \sum_{k\leq k^*} m_{0k}\lambda_k \exp(-\eta \lambda_k T^{1-\gamma}) + \sum_{k > k^*} m_{0k}\lambda_k \exp(-\eta \lambda_k T^{1-\gamma})\\
&\leq T^{-1}\|w^*\|^2_{\Lambda_{1:k^*}} + \|w^*\|^2_{\Lambda_{k^*:\infty}}
\end{align*}
\begin{align*}
variance_{T+1} &\lesssim \sum_k \eta \sigma^2\lambda_k^2 T^{1-2\gamma}\exp\left(-\frac{1}{2}(1-\gamma)\eta\lambda_k T^{1-\gamma}\right) + \sum_k  \frac{\sigma^2\lambda_k}{1-\gamma }  T^{-\gamma}\\
&\lesssim \sum_k \frac{\sigma^2 \lambda_k}{1-\gamma} T^{-\gamma} + \sum_k  \frac{\sigma^2\lambda_k}{1-\gamma }  T^{-\gamma} & \text{since } xe^{-ax} \leq \frac{1}{ae}\\
&\lesssim  \sum_k \frac{\sigma^2 \lambda_k}{1-\gamma} T^{-\gamma} 
% &\lesssim \sum_{k\leq k^*} \eta \sigma^2\lambda_k^2 T^{1-2\gamma} T^{-\frac{1-\gamma}{2}}+ \sum_{k > k^*}  \sigma^2\lambda_k \ln(T) T^{-\gamma}  
% + \sum_{k}  \frac{\sigma^2\lambda_k}{1-\gamma }  T^{-\gamma}
\end{align*}

So $\mathcal R(w_T) - \sigma^2/2 \lesssim T^{-1}\|w^*\|^2_{\Lambda_{1:k^*}} + \|w^*\|^2_{\Lambda_{k^*:\infty}} + \sum_k \frac{\sigma^2 \lambda_k}{1-\gamma} T^{-\gamma} $
